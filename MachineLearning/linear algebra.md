  $$
  \triangledown f(x)= grad f(x)=\frac{\partial f(x)}{\partial x} = 
  \begin{pmatrix}
  \frac{\partial f(x)}{\partial x_1} \\ \frac{\partial f(x)}{\partial x_2} \\.\\.\\.\\\frac{\partial f(x)}{\partial x_d}
  \end{pmatrix}
  $$
